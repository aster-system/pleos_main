<new_definition name="topological_boundary">
    <content>la frontière d'un ouvert est la partie de l'ensemble contenant cette ouvert qui limite les éléments à l'intérieur de l'ouvert de ceux à l'extérieur de l'ouvert</content>
</new_definition>
<new_definition name="topological_closed_set">
    <content>un fermé est une partie d'un ensemble tel que son complémentaire dans l'ensemble soit un ouvert</content>
</new_definition>
<new_definition name="topological_closure">
    <content>un fermé d'une partie "F" d'un ensemble "E" est une partie de "E" content "F" et la frontière de "F"</content>
</new_definition>
<new_definition name="topological_neighbourhood">
    <content>le voisinage d'un point "a" d'un ensemble d'un espace topologique est une partie de cette ensemble qui inclut un ouvert auquel appartient "a"</content>
</new_definition>
<new_definition name="topological_open_set">
    <content>un ouvert est une partie d'un ensemble qui ne contient aucun élément de sa frontière</content>
</new_definition>
<new_definition name="topological_space">
    <content>un espace topologique représente l'étude d'un ensemble "E" avec une topologie "T" associée</content>
</new_definition>
<new_definition name="topology">
    <content>la topologie est l'étude des espaces topologiques et de leurs propriétés possibles</content>
</new_definition>
