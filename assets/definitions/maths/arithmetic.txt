<new_definition name="arithmetic">
    <content>l'arithmétique est la branche des mathématiques étudiant les nombres communs (entiers, rationnels et réels)</content>
</new_definition>
<new_definition name="arithmetic_congruence">
    <content>deux nombres entiers sont dit congrus modulo "n" (avec n un autre entier) si ils ont le même reste par la division par "n"</content>
</new_definition>
<new_definition name="operation">
    <content>une opération est une action où, gràce à une (ou plusieurs) valeur de départ (nommée opérande), on y définit une égalité représentant une (ou plusieurs) valeurs traitées via les valeurs de départs</content>
</new_definition>

<new_definition name="natural_number">
    <content name="addition">une addition entre deux nombres entiers naturels représente une opération entre deux nombres entiers naturels "a" et "b", donnant comme valeur la valeur partant de "a" obtenue après la répétition de "b" fois l'objet "successeur"</content>
    <content name="bigger">un nombre entier naturel "a" est dit strictement supérieur à un nombre entier naturel "b" si nous pouvons obtenir "a" en appliquant une ou plusieurs fois la fonction successeur (en additionnant 1) à "b"</content>
    <content name="bigger_equal">un nombre entier naturel "a" est dit supérieur ou égal à un nombre entier naturel "b" si "a" est supérieur à "b" ou si "a" est égal à "b"</content>
    <content name="multiplication">une multiplication entre deux nombres entiers naturels représente une opération entre deux nombres entiers naturels "a" et "b", donnant comme valeur la valeur partant de "a" obtenue après la répétition de "b" fois l'opération d'addition de "a" avec lui même</content>
</new_definition>

<new_definition name="real_number">
    <content>un nombre réel représente un nombre constitué d'une partie entière (représentant un nombre naturel) et d'une partie décimale quelconque (qui peut être infinie et non-périodique)</content>
</new_definition>
<new_definition name="real_numbers_set">
    <content>l'ensemble des nombres réels représente l'ensemble contenant TOUS les nombres réels, lui permettant d'être totalement ordonné, complet (et donc connexe), et d'avoir la propriété de la borne supérieure</content>
</new_definition>

<new_definition name="function_holomorphic">
    <content>une fonction est dite holomorphe dans un sous-ensemble E de C (voir, dans tout C, ou sur un point unique) si elle est définie et dérivable dans tout E</content>
</new_definition>
