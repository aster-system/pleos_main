<new_definition name="inference">
    <content>une inférence est une façon de démontrer logiquement une proposition (nommée "conclusion"), en s'appuyant sur d'autres propositions appelées "prémisses"</content>
</new_definition>
<new_definition name="logic_abduction">
    <content>une abduction est une inférence représentant le passage de deux propositions insuffisantes pour en tirer une déduction, à une proposition permettant de réaliser une hypothèse de déduction</content>
</new_definition>
<new_definition name="logic_deduction">
    <content>une déduction est une inférence représentant le passage de deux propositions obligatoirement vraies vers une proposition, dont on déduit quelle est vraie</content>
</new_definition>
<new_definition name="logic_induction">
    <content>une induction est une inférence représentant le passage de deux propositions probablement vraies vers une proposition, dont on déduit quelle est probablement vraie</content>
</new_definition>
<new_definition name="syllogism">
    <content>un syllogisme est une façon de démontrer logiquement un propos (nommé "conclusion"), en s'appuyant sur d'autres propos appelés "prémisses"</content>
</new_definition>

<new_definition name="axiom">
    <content>les axiomes sont des propositions théoriquement indémontrées (mais qui, en pratique, sont tout à fait """"logiques""""", en accord avec les autres sciences et ne nécessitent pas de démonstration) qui servent de bases à toutes les mathématiques</content>
</new_definition>
<new_definition name="demonstration">
    <content>une démonstration mathématique représente le chemin permettant d'aller des axiomes au théorème recherché</content>
</new_definition>
<new_definition name="prenex_form">
    <content>une formule mathématique est dite prénexe si tous ces quantificateurs se situent avec la proposition de la formule</content>
</new_definition>
<new_definition name="quantifier">
    <content>un quantificateur est un objet mathématique utilisable pour poser le contexte de vérité d'une proposition (dans un / des ensembles précis)</content>
</new_definition>




<new_definition name="automorphism">
    <content>un automorphisme est un morphisme, qui est en même temps un endomorphisme et un isomorphisme</content>
</new_definition>
<new_definition name="automorphism_group">
    <content>un groupe d'automorphisme d'une structure E est l'ensemble des automorphismes possibles sur la structure E</content>
</new_definition>
<new_definition name="endomorphism">
    <content>un endomorphisme est un morphisme d'une structure mathématique E, vers cette même structure E en image</content>
</new_definition>
<new_definition name="morphism">
    <content>un morphisme est une application entre deux ensembles de même structures, où la structure "d'arrivée" possède les mêmes "propriétés" que la structure de départ, avec possibilité de les interpréter différement</content>
</new_definition>
<new_definition name="morphism_kernel">
    <content>le noyau d'un morphisme de groupe est l'ensemble du cors de départ du morphisme qui donne en image l'élément neutre du corps d'arrivée</content>
</new_definition>
<new_definition name="morphism_ordered_set">
    <content>un morphisme entre deux ensembles ordonnées est un morphisme de ces deux ensembles, où la structure d'ordre est préservée (x &lt; g implique f(x) &lt; f(g))</content>
</new_definition>
<new_definition name="isomorphism">
    <content>un isomorphisme est un morphisme où il existe un morphisme inverse (un autre isomorphisme), permettant de retrouver un élément du premier ensemble grâce à un élément du second</content>
    <content type="theorem" name="1">il est possible d'extraire un isomorphisme de tous les morphismes "f" possibles, sous la forme de l'isomorphisme de l'ensemble de départ quotienté par le noyau de "f" vers l'image de "f"</content>
</new_definition>
