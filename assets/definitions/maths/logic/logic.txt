<new_definition name="logic">
    <content>la logique représente l'ensemble des outils permettant de raisonner, et donc de justifier des propos</content>
    <content name="classic">la logique classique est la première tentative de "parfaitement" formaliser la logique en mathématique au XIXème siècle</content>
    <content name="propositions">la logique des propositions (aussi nommé "calcul des propositions") est une forme de logique étudiant les propositions et les relations entre les propositions</content>
</new_definition>
<new_definition name="predicate">
    <content>un prédicat est un schéma de proposition utilisant une variable externe, qui permet d'obtenir une proposition concrète (soit vraie soit fausse) selon la valeur précise de la variable</content>
</new_definition>
<new_definition name="proposition">
    <content>une proposition logique est une information quelconque, possédant une valeur de vérité (elle est soit vraie, soit fausse)</content>
</new_definition>
<new_definition name="semantic">
    <content>la sémantique représente un moyen de donner un sens à une proposition énoncé grâce à une syntaxe</content>
</new_definition>
<new_definition name="syntax">
    <content>une syntaxe représente les règles qui doivent suivre des mots (ou n'importe quel objet qu'on peut "écrire") rendant une phrase correcte</content>
</new_definition>
<new_definition name="verity_table">
    <content>une table de vérité est un moyen de représenter les valeurs de vérités une proposition composée selon chaque valeur possibles de chacune de ses propositions composantes</content>
</new_definition>

<new_definition name="excluded_middle">
    <content>le principe du tiers-exclu est un principe logique qui stipule que une proposition NE PEUT QU'ÊTRE vraie ou fausse, mais pas les deux en même temps</content>
</new_definition>
<new_definition name="inference">
    <content>une inférence est une façon de démontrer logiquement une proposition (nommée "conclusion"), en s'appuyant sur d'autres propositions appelées "prémisses"</content>
</new_definition>
<new_definition name="logic_abduction">
    <content>une abduction est une inférence représentant le passage de deux propositions insuffisantes pour en tirer une déduction, à une proposition permettant de réaliser une hypothèse de déduction</content>
</new_definition>
<new_definition name="logic_deduction">
    <content>une déduction est une inférence représentant le passage de deux propositions obligatoirement vraies vers une proposition, dont on déduit quelle est vraie</content>
</new_definition>
<new_definition name="logic_induction">
    <content>une induction est une inférence représentant le passage de deux propositions probablement vraies vers une proposition, dont on déduit quelle est probablement vraie</content>
</new_definition>
<new_definition name="logical_connective">
    <content>un connecteur logique est un moyen d'écrire une proposition complexe en coupant cette proposition en "sous-propositions", et d'analyser la proposition de départ en analysant ces sous-propositions</content>
    <content name="and">le connecteur logique "et" permet de lier deux propositions en une proposition finale "P", qui n'est vrai QUE SI les deux propositions de bases ("proposition 1" ET "proposition 2") sont vraies (sinon, elle est fausse)</content>
    <content name="contraposition">la contraposée est un moyen de prouver une implication P "A implique B", en prouvant que "pas A implique pas B"</content>
    <content name="equivalence">le connecteur logique "équivaut" permet de lier deux propositions en une proposition finale "P", qui n'est vraie QUE SI les deux propositions de bases sont toutes les deux vraies ou toutes les deux fausses (elles ont l'exact même valeur de vérité)</content>
    <content name="implication">le connecteur logique "implique" permet de lier deux propositions en une proposition finale "P", qui, si "A implique B" n'est vrai QUE SI la proposition A n'est vraie QUE QUAND la proposition B l'est aussi</content>
    <content name="or">le connecteur logique "ou" permet de lier deux propositions en une proposition finale "P", qui n'est vrai QUE SI l'une des deux propositions de bases ("proposition 1" OU "proposition 2") est vraie, ou si les deux sont vraies (sinon, elle est fausse)</content>
</new_definition>
<new_definition name="logical_negation">
    <content>la négation d'une proposition A représente une autre proposition qui est égale à "A est fausse", aussi notée "pas A" ou "non A"</content>
</new_definition>
<new_definition name="modus_ponens">
    <content>le modus ponens est est une loi logique affirmant que si "A implique B" et "A" est vraie, alors "B" est vraie</content>
</new_definition>
<new_definition name="modus_tollens">
    <content>le modus ponens est est une loi logique affirmant que si "A implique B" et "B" est fausse, alors "A" est fausse</content>
</new_definition>
<new_definition name="syllogism">
    <content>un syllogisme est une façon de démontrer logiquement un propos (nommé "conclusion"), en s'appuyant sur d'autres propos appelés "prémisses"</content>
</new_definition>

<new_definition name="mathematical_induction">
    <content>le raisonnement par récurrence est un principe de raisonnement permettant de prouver que si un prédicat "P(n)" est vraie pour "P(a)" et que, pour tout élément "k" d'un ensemble "N", "P(k) implique p(successeur(k))", alors "P(n)" est vraie pour tout "n" supérieur à "a"</content>
</new_definition>

<new_definition name="axiom">
    <content>les axiomes sont des propositions théoriquement indémontrées (mais qui, en pratique, sont tout à fait """"logiques""""" / évidentes, en accord avec les autres sciences et ne nécessitent pas de démonstration) qui servent de bases à toutes les mathématiques</content>
    <content name="schema">les schémas d'axiomes sont un moyen de définir un "plan" (ou un schéma) pour créer une quantité infinie d'axiome</content>
</new_definition>
<new_definition name="axiom_theory">
    <content>une théorie axiomatique est un ensemble d'axiomes, permettant de déduire / démontrer énormément de propositions (incluses dans la théorie)</content>
    <content name="consistency">une théorie axiomatique est dite cohérente (ou consistante) si elle ne contient pas de contradiction logique (une contradiction serait une proposition "A est vrai et A n'est pas vrai")</content>
</new_definition>
<new_definition name="demonstration">
    <content>une démonstration mathématique représente le chemin permettant d'aller des axiomes au théorème recherché</content>
</new_definition>
<new_definition name="prenex_form">
    <content>une formule mathématique est dite prénexe si tous ces quantificateurs se situent avec la proposition de la formule</content>
</new_definition>
<new_definition name="quantifier">
    <content>un quantificateur est un objet mathématique utilisable pour poser le contexte d'utilisation de la variable d'un prédicat (la transformant en proposition)</content>
    <content name="existential">le quantificateur existentiel transforme un prédicat en une proposition vraie si il existe AU MOINS UNE proposition vraie formulable avec ce prédicat</content>
    <content name="universal">le quantificateur universel transforme un prédicat en une proposition vraie si TOUTES les propositions formulables avec ce prédicat sont vraie</content>
</new_definition>




<new_definition name="automorphism">
    <content>un automorphisme est un morphisme, qui est en même temps un endomorphisme et un isomorphisme</content>
</new_definition>
<new_definition name="automorphism_group">
    <content>un groupe d'automorphisme d'une structure E est l'ensemble des automorphismes possibles sur la structure E</content>
</new_definition>
<new_definition name="endomorphism">
    <content>un endomorphisme est un morphisme d'une structure mathématique E, vers cette même structure E en image</content>
</new_definition>
<new_definition name="morphism">
    <content>un morphisme est une application entre deux ensembles de même structures, où la structure "d'arrivée" possède les mêmes "propriétés" que la structure de départ, avec possibilité de les interpréter différement</content>
</new_definition>
<new_definition name="morphism_kernel">
    <content>le noyau d'un morphisme de groupe est l'ensemble du cors de départ du morphisme qui donne en image l'élément neutre du corps d'arrivée</content>
</new_definition>
<new_definition name="morphism_ordered_set">
    <content>un morphisme entre deux ensembles ordonnées est un morphisme de ces deux ensembles, où la structure d'ordre est préservée (x &lt; g implique f(x) &lt; f(g))</content>
</new_definition>
<new_definition name="isomorphism">
    <content>un isomorphisme est un morphisme où il existe un morphisme inverse (un autre isomorphisme), permettant de retrouver un élément du premier ensemble grâce à un élément du second</content>
    <content type="theorem" name="1">il est possible d'extraire un isomorphisme de tous les morphismes "f" possibles, sous la forme de l'isomorphisme de l'ensemble de départ quotienté par le noyau de "f" vers l'image de "f"</content>
</new_definition>
