<new_definition name="differential_calculus">
    <content>le calcul différentiel est une branche du calcul infinitésimal où les valeurs infinitésimales servent à étudier les variations de fonctions</content>
    <content name="fluxions">les fluxions représentent la vitesse à laquelle une fonction monte / descend en un point quelconque</content>
</new_definition>
<new_definition name="infinitesimal_calculus">
    <content>le calcul infinitésimal est une branche du calcul mathématique s'intéressant à des calculs s'approchant de plus en plus de certaines valeurs</content>
</new_definition>
<new_definition name="integration_calculus">
    <content>le calcul intégral est une branche du calcul infinitésimal où les valeurs infinitésimales servent à étudier une fonction comme les variations infinitésimales d'autres fonctions</content>
</new_definition>

<new_definition name="analysis_continuity">
    <content>la continuité mathématique représente un concept suggérant la possibilité de se rapprocher indéfiniment d'une valeur dans un espace, quelque soit la façon dont vous faites</content>
    <content name="function_real">une fonction réelle est continue en une valeur "a" si la valeur de sa limite à droite en "a" est égale à la valeur de sa limite à gauche en "a", et que cette limite est finie</content>
</new_definition>
<new_definition name="analysis_interval">
    <content>une intervalle est un ensemble de nombre réel représentant tous les nombres réels situés entre deux autres nombres réels</content>
    <content name="infimum">la borne inférieure d'une partie "F" d'un ensemble "E" est le plus grand minorant de "F" (et donc le plus grand élément inférieur à tous les éléments de "F")</content>
    <content name="majorant">un majorant d'une partie "F" d'un ensemble "E" est un élément de "E" supérieur à tous les éléments de "F"</content>
    <content name="minorant">un minorant d'une partie "F" d'un ensemble "E" est un élément de "E" inférieur à tous les éléments de "F"</content>
    <content name="supremum">la borne supérieure d'une partie "F" d'un ensemble "E" est le plus petit majorant de "F" (et donc le plus petit élément supérieur à tous les éléments de "F")</content>
</new_definition>
<new_definition name="analysis_real_function">
    <content>une fonction réelle est une fonction allant de l'ensemble "R" vers l'ensemble "R"</content>
</new_definition>
<new_definition name="analysis_sequence">
    <content>une suite mathématique est une fonction mathématique allant de "N" à n'importe quel ensemble possible</content>
    <content name="convergent">une suite mathématique est dite convergente si sa limite est une valeur finie "v" (on dit que la suite converge vers "v")</content>
    <content name="negligible">une suite mathématique "u" est dite négligeable devant une suite "v" lorsque "v" se rapproche plus vite de sa limite que "u"</content>
</new_definition>
<new_definition name="analysis_limit">
    <content>une limite mathématique représente en quelque sorte une valeur d'on se rapproche indéfiniment quand, dans une fonction (admettant comme ensemble d'arrivée un espace métrique), on se rapproche indéfiniment d'un point précis (sans jamais réellement l'atteindre)</content>
    <content name="rigorous">une limite mathématique de fonction en une valeur "a" est une valeur "v" telle que, pour tout nombre réel possible "x", il existe toujours un nombre réel "b" (supérieur à "a" si on calcul la limite à droite, et inférieur sinon) tel que toute valeur entre "a" et "b" admet comme image une distance à "v" inférieure au nombre "x"</content>
    <content name="rigorous_infinite">une limite mathématique infinie de fonction en une valeur "a" est l'un des deux infinies ("+ l'infini" ou "- l'infini") tel que, pour tout nombre réel possible "x", il existe toujours un nombre réel "b" (supérieur à "a" si on calcul la limite à droite, et inférieur sinon) tel que toute valeur entre "a" que "b" admet comme image une valeur supérieure au nombre "x"</content>
    <content name="sequence">une limite mathématique de suite représente en quelque sorte la valeur dont on se rapproche indéfiniment quand, dans une suite (admettant comme ensemble d'arrivée un espace métrique), on tend "n" vers "+ l'infini" (sans jamais réellement l'atteindre)</content>
    <content name="sequence_rigorous">une limite mathématique de suite vers une valeur "v" est une valeur "v" telle que, pour tous nombre réel possible "x", il existe toujours un nombre entier naturel "a" tel que la distance entre n'importe quelle valeur de la suite à partir de ce nombre "a" et le nombre "v" sont inférieurs à "x"</content>
    <content name="sequence_rigorous_infinite">une limite mathématique de suite vers une valeur infinie est un des deux infinis ("+ l'infini" ou "- l'infini") tel que, pour tous nombre réel possible "x", il existe toujours un nombre entier naturel "a" tel que toutes valeurs de la suite à partir de ce nombre "a" est supérieure (si la limite est "+ l'infini") ou inférieur (si la limite est "- l'infini") à "x"</content>
</new_definition>
<new_definition name="function_graph">
    <content>un graphe de fonction représente l'ensemble des couples formés par (dans l'ordre) chaque élément de l'ensemble du domaine de définition de la fonction et de son image par cette fonction</content>
</new_definition>

<new_definition name="function_space">
    <content>un espace fonctionnel est un ensemble d'applications (et donc, de fonctions) entre deux ensembles X et Y</content>
</new_definition>

<new_definition name="analysis_limit_">
    <content>une limite mathématique est un outil / un opérateur permettant d'approximer une "infinité de fois" une valeur</content>
    <content name="sequence">une limite de suite est la valeur vers laquelle on se rapproche si on prend des valeurs "toujours plus proches de l'infini"</content>
    <content name="function">une limite de fonction en une valeur "v" est la valeur vers laquelle on se rapproche si on prend des valeurs "toujours plus proches de v"</content>
</new_definition>

<new_definition name="metric_space">
    <content>un espace métrique est un couple représentant un ensemble quelconque, et une opération binaire sur cet ensemble pouvant s'interpréter comme la distance entre deux éléments quelconques de l'ensemble</content>
    <content name="distance">une distance est une loi définie sur un ensemble "E", allant de E X E vers R+, où la loi est symétrique, séparée et respecte l'inégalité triangulaire</content>
</new_definition>
<new_definition name="metric_space_completude">
    <content>un espace métrique est dit complet si, pour toutes les valeurs possibles dans l'ensemble de l'espace, il est possible de s'approcher aussi près que l'on veut de cet ensemble (en réduisant petiit à petit la distance entre une valeur plus simple et la valeur que l'on veut approcher)</content>
</new_definition>
