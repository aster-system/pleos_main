<new_definition name="algebra">
    <content name="classic">l'algèbre classique est la discipline mathématique qui étudie le concept brut de calcul algébrique</content>
    <content name="general">l'algèbre générale est la discipline mathématique qui étudie les différentes formes d'algèbres classiques possibles, sous forme de structures algébriques</content>
</new_definition>
<new_definition name="algebric_structure">
    <content>une structure algébrique est la mise en commun d'un ensemble et d'une (ou plusieurs) opération de calcul (nommées "lois de composition internes"), pour les étudier de manière conjointe</content>
</new_definition>
<new_definition name="calculus">
    <content>un calcul est une suite d'actions qui permettent de passer d'une forme quelconque à une autre</content>
    <content name="algebric">un calcul est dit algébrique si il permet de travailler avec des indéterminés</content>
    <content name="numeric">un calcul est dit numérique quand il ne permet de travailler que sur des nombres connus</content>
</new_definition>
<new_definition name="cayley_table">
    <content>une table de Cayley est une représentation sous forme de tableau à deux entrées des valeurs obtenues grâce à une loi de composition entre chaque élément d'une structure algébrique finie</content>
</new_definition>
<new_definition name="composition_law">
    <content>une loi de composition est une application entre tous les éléments de deux ensembles "A" et "B", et vers des éléments de "A"</content>
    <content name="internal">une loi de composition est dite interne lorsque les deux éléments de départs sont les mêmes (et donc, les mêmes que l'ensemble d'arrivée aussi)</content>
    <content name="rigorous">une loi de composition est une application entre tous les éléments du produit cartésien de deux ensembles "A" et "B" (ou du produit cartésien de "B" et "A"), et vers des éléments de "A"</content>
</new_definition>
<new_definition name="operation">
    <content>une opération est une action où, grâce à une (ou plusieurs) valeur de départ (nommée opérande), on y définit une égalité représentant une (ou plusieurs) valeurs traitées via les valeurs de départs</content>
    <content name="stability">une opération entre plusieurs objets est dite stable (ou close) en un ensemble E si, pour tous les éléments possibles au départ, l'arrivée est toujours dans E</content>
</new_definition>

<new_definition name="algebra_root">
    <content>la racine n-ième d'un nombre "m" représente le nombre qui vaut m si il est mit à l'exposant n</content>
</new_definition>
<new_definition name="unity_root">
    <content>une racine de l'unité est un nombre complexe tel que son carré vaut 1 (en effet, dans l'algèbre des complexes, d'autres nombres que 1 ont un carré valant 1)</content>
</new_definition>

<new_definition name="algebra_archimedean">
    <content>une structure algébrique est dite "archimédienne" si, pour tout nombre "a" et "b", il existe un multiple de "c" tel que "a * c" est plus grand que "b"</content>
    <content name="magma">un magma (et donc, un monoïde ou un groupe) est dit "archimédien" si, pour toute valeur "a" et "b", il existe un nombre entier naturel de "c" tel que "c" répétition de la loi sur "a" est plus grand que "b"</content>
</new_definition>
<new_definition name="algebra_associativity">
    <content>une loi de composition interne est dite associative si l'ordre de réalisation de plusieurs opérations ne modifie pas le résultat<content>
</new_definition>
<new_definition name="algebra_distributivity">
    <content>une opération "." est dite distributive par rapport à une opération "~" lorsqu'il est possible de transformer l'opération "a . (b ~ c)" en l'opération "a . b ~ a . c"</content>
</new_definition>
<new_definition name="algebra_generated_set">
    <content name="group">un sous-groupe engendré d'une partie "F" d'un groupe "G" sur un ensemble "E" représente l'ensemble des éléments de "G" que nous pouvons obtenir avec seulement une répétition de la loi du groupe sur des éléments de cette partie "F" (et / ou de leurs inverses)</content>
</new_definition>
<new_definition name="algebra_generating_set">
    <content name="group">une partie génératrice d'un groupe est une partie "F" d'une groupe "G" sur un ensemble "E" où le sous-groupe engendré par "F" représente l’entièreté de "G"</content>
</new_definition>
<new_definition name="algebra_neutral_element">
    <content>un élément est dit neutre par une loi de composition si l'utilisation de la loi sur un autre élément "a" et cet élément donne "a"</content>
</new_definition>

<new_definition name="algebra_field">
    <content>un corps est un anneau où chaque élement (non-nul) possède un inverse par la deuxième opération</content>
</new_definition>
<new_definition name="algebra_group">
    <content>un groupe est un magma dont la loi de composition interne et associative, admet un neutre, et est inversible</content>
    <content name="abelian">un groupe est dit abélien si sa loi de composition est commutative</content>
</new_definition>
<new_definition name="algebra_group_znz">
    <content>le groupe additif "Z/nZ" est un sous-groupe de "(Z, +)", où l'ensemble du groupe représente "Z" quotienté par la relation d'équivalence qui lie les éléments ayant le même reste par la division euclidienne par "n"</content>
</new_definition>
<new_definition name="algebra_group_theory">
    <content>la théorie des groupes représente la discipline qui étudie les groupes algébriques de manière très poussée</content>
</new_definition>
<new_definition name="algebra_magma">
    <content>un magma est une structure algébrique représentant un ensemble quelconque et une seule loi de composition interne sur cette ensemble, sans plus de détail</content>
</new_definition>
<new_definition name="algebra_order">
    <content name="element">l'ordre d'un élément d'un groupe représente le cardinal du sous-groupe engendré par cet élément</content>
    <content name="group">l'ordre d'un groupe représente le cardinal de l'ensemble qui constitue le groupe</content>
</new_definition>
<new_definition name="algebra_ring">
    <content>un anneau (aussi nommé anneau unitaire) est une structure algébrique définie sur un ensemble, avec deux lois de compositions internes : une formant un groupe abélien, et une formant un monoïde, distributive par rapport à la première</content>
</new_definition>
<new_definition name="algebra_subgroup">
    <content>un sous-groupe H d'un groupe G est un groupe sur un sous-ensemble de l'ensemble de G, muni de la même loi de composition que G</content>
    <content name="class">une classe à gauche (ou à droite) d'un élément "a" suivant un sous-groupe représente l'ensemble des modifications possibles dans un sous-groupe, en ne faisant qu'utiliser la loi de composition à gauche (ou à droite) des éléments du sous-groupe avec l'élément "a" du groupe</content>
    <content name="index">l'indice d'un sous-groupe représente le cardinal des classes à gauche (ou à droite) d'un sous-ensemble</content>
</new_definition>
<new_definition name="algebra_subring">
    <content>un sous-anneau A d'un anneau B est un anneau sur un sous-ensemble de l'ensemble de A, où l'addition forme un sous-groupe du groupe additif de "A", où "B" est stable par la multiplication et où le neutre de la multiplication de "A" se situe dans "B"</content>
</new_definition>
<new_definition name="vector_space">
    <content>un espace vectoriel est une structure algébrique construire à partir d'un corps, sur un ensemble quelconque (qui n'est pas forcément celui du corps), tel qu'il contient une opération (généralement une forme d'addition) formant un groupe abélien sur l'espace vectoriel et qu'il existe une loi externe à gauche sur l'espace vectoriel avec les éléments du corps (qui doit être associative, admettre un neutre et distributive à gauche par rapport à la première loi de l'espace vectoriel)</content>
</new_definition>

<new_definition name="algebra_center">
    <content>le centre d'une structure algébrique représentent l'ensemble des éléments de cette structure commutant avec tous les autres éléments</content>
</new_definition>
<new_definition name="algebra_group_action">
    <content>une action de groupe est un moyen d'opérer un groupe avec un ensemble "E", de tel manière que tous les résultats de ces opérations appartiennent à "E"</content>
</new_definition>
<new_definition name="algebra_ideal">
    <content>un idéal I d'un anneau représente un sous-groupe de son groupe additif, où la multiplication de chacun de ces éléments par un élément du corps de base donne un élément de I (l'idéal est dit stable par la multiplication)</content>
</new_definition>
<new_definition name="algebra_rank">
    <content>le rang d'une famille de vecteurs d'un espace vectoriel représente la dimension du sous-espace engendré par ce vecteur</content>
    <content name="linear_application">le rang d'une application linéaire sur un espace vectoriel est la dimension de l'image de cette application</content>
</new_definition>
<new_definition name="algebra_subfield">
    <content>un sous-corps K d'un corps L est un corps sur un sous-ensemble de l'ensemble de L, dans lequel la structure de corps est conservée (présence d'un neutre, le même que L, et toutes les opérations dans K, les mêmes que dans L, donnent un élément de K)</content>
</new_definition>
<new_definition name="algebra_subgroup_proper">
    <content>un sous-groupe H d'un groupe G est dit propre si H n'est pas un groupe trivial de G</content>
</new_definition>
<new_definition name="algebra_subgroup_trivial">
    <content>le sous-groupe trivial d'un groupe G est un sous-groupe de G sur l'ensemble ne contenant que l'élément neutre de G</content>
</new_definition>
<new_definition name="morphism">
    <content>un morphisme de structures est une application entre deux ensembles de deux structures, où les lois sont compatibles entre elles après utilisation de l'application</content>
    <content name="algebra">un morphisme de structures algébriques est une application entre deux ensembles de deux structures algébriques, où l'on doit s'assurer que les lois de composition de la première structure ont les mêmes propriétés que les lois de composition de la deuxième structure après utilisation de l'application, et que les lois de composition sont compatibles entre elles après utilisation de l'application</content>
</new_definition>

<new_definition name="cayley_theorem">
    <content>le théorème de Cayley est un théorème qui affirme que tout groupe "G" est isomorphe à un groupe de permutation de l'ensemble de "G"</content>
</new_definition>
<new_definition name="symmetric_group">
    <content>le groupe symétrique (ou groupe des permutations) d'un ensemble "E" est un groupe sur l'ensemble des permutations de "E", et utilisant comme loi de composition la composition de fonction</content>
    <content name="permutation_group">un groupe de permutation d'un ensemble "E" est un sous-groupe du groupe symétrique (aussi nommé groupe des permutations) de "E"</content>
</new_definition>

<new_definition name="algebra_lagrange_theorem">
    <content>le théorème de Lagrange formule que l'ordre d'un groupe G (le cardinal de son ensemble) est divisible par ceux de ses sous-groupes H</content>
</new_definition>
<new_definition name="wedderburn_theorem">
    <content>le théorème de Wedderburn est un théorème d'algèbre, affirmant que un corps fini (d'ensemble fini) est obligatoirement commutatif</content>
</new_definition>

<new_definition name="complex_number">
    <content>un nombre complexe est un nombre représentant la somme de deux nombres d'unités différentes : un par l'unité "1" (nommé nombre réel, car étant un nombre réel) et un par l'unité "1" (nommé imaginaire, car étant un nombre imaginaire)</content>
</new_definition>
<new_definition name="imaginary_number">
    <content>un nombre imaginaire est un nombre réel quelconque multiplié à une autre unité que "1" : l'unité "i" (définie comme le nombre dont le carré vaut -1)</content>
</new_definition>
