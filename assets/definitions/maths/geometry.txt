<new_definition name="geometry">
    <content>la géométrie est historiquement la discipline mathématique qui étudie les possibilités sur des plans 2D (sur papier) et 3D (dans l'espace / la vraie vie)</content>
    <content name="analytic">la géométrie analytique est une façon précise d'étudier la géométrie, en utilisant le concept de coordonnées</content>
</new_definition>

<new_definition name="al_kashi_theorem">
    <content>le théorème d'Al-Kashi (ou loi des cosinus) est un théorème établissant une relation entre deux longueurs de côtés et la mesure d'un angle (ou la mesure de deux angles et la longueur d'un côté) dans un triangle quelconque</content>
    <content type="theorem"><math><mforall><mi>A, B, C</mi><min><mi>R</mi><msup>3</msup><mo>,</mo><mi>[AB]²</mi><mo>=</mo><mi>[BC]²</mi><mo>+</mo><mi>[AC]²</mi><mo>-</mo><mi>2</mi><mo>*</mo><mi>[BC]</mi><mo>*</mo><mi>[AC]</mi><mo>*</mo><mi>cos(ACB)</mi></math></content>
</new_definition>
<new_definition name="angle">
    <content>un angle entre 3 points est une mesure de la différence d'orientation entre les deux droites formées par ces deux points (en sachant que nous travaillons sur les droites passant toutes les deux par un seul des points de départ)</content>
    <content name="right">un angle "ABC" est dit droit en "B" quand les droites formés par "AB" et "BC" sont parfaitement perpendiculaire</content>
</new_definition>
<new_definition name="circle">
    <content>un cercle est un ensemble de point, tel que tous les points de l'ensemble se trouvent à une distance similaire (appelée "rayon" du cercle) d'un même point, dit "centre" du cercle</content>
</new_definition>
<new_definition name="ellipse">
    <content>une ellipse est une figure géométrique 2D résultant de la projection d'un cône 3D sur un plan 2D passant par son axe de rotation, sans passer par le sommet, ni être coplanaire à la base du cône</content>
</new_definition>
<new_definition name="hypotenuse">
    <content>l’hypoténuse d'un triangle rectangle est le côté opposé à l'angle droit du rectangle</content>
</new_definition>
<new_definition name="length">
    <content>une longueur entre deux points est une représentation sous forme de nombre de la distance entre ces deux points</content>
</new_definition>
<new_definition name="polygon">
    <content>un polygone est une figure géométrique formé par un nombre fini de segments reliés entre eux</content>
</new_definition>
<new_definition name="pythagorean_theorem">
    <theorem>le théorème de Pythagore stipule que le carré de la longueur de l’hypoténuse (dans un triangle rectangle) est égal à la somme des carrés des deux autres côtés du triangle</theorem>
    <theorem name="converse">la réciproque du théorème de Pythagore stipule que si le carré de la longueur d'un des côtés du triangle est égal à la somme des carrés des deux autres côtés du triangle, alors ce triangle est rectangle, et le côté le plus grand est l'hypoténuse du triangle</theorem>
</new_definition>
<new_definition name="tangent">
    <content>la fonction tangente représente le coefficient directeur de la droite tangente au cercle en un certain point, à un certain angle "a"</content>
</new_definition>
<new_definition name="triangle">
    <content>un triangle est un polygône à 3 côtés</content>
    <content name="rectangle">un triangle est dit rectangle si il contient un angle droit quelconque</content>
</new_definition>
<new_definition name="trigonometric_circle">
    <content>le cercle trigonométrique est un cercle de rayon 1 et de centre le repère du plan, permettant de facilement définir les fonctions trigonométriques</content>
</new_definition>
<new_definition name="trigonometric_function">
    <content>une fonction trigonométrique est une fonction permettant de mettre en relation des longueurs de côtés et des angles dans un triangle</content>
</new_definition>
<new_definition name="trigonometry">
    <content>la trigonométrie est la discipline mathématique qui traite des mesures dans un triangle ("trigono" vient de "triangle", et "métrie" vient de "mesure")</content>
</new_definition>

<new_definition name="vectorial_space_dimension">
    <content>la dimension d'un espace vectoriel représente le nombre d'informations (provenant de son corps de base) nécessaire pour décrire tous les vecteurs de l'espace</content>
    <content name="rigorous">la dimension (aussi appelée dimension de Hamel) d'un espace vectoriel est le cardinal (commun) de ces bases</content>
</new_definition>

<new_definition name="hermitian_scalar_product">
    <content>un produit scalaire hermitien est un produit scalaire adapté au corps des nombres complexes (prenant deux vecteurs complexe et donnant un nombre complexe), à quelques différences prêt : il est sesquilinéaire et possède la symétrie hermitienne</content>
</new_definition>
<new_definition name="hermitian_space">
    <content>un espace hermitien est une espace vectoriel sur le corps (commutatif) des nombres complexes, muni d'un produit scalaire "hermitien"</content>
</new_definition>
<new_definition name="sesquilinearity">
    <content>une application sesquilinéaire représente une application linéaire à droite et semi-linéaire à gauche (où la multiplication par un scalaire se fait via son conjugué)</content>
</new_definition>

<new_definition name="hilbert_base">
    <content>une base de Hilbert est une version généralisée des bases d'espaces euclidiens à un espace de Hilbert à "n" dimensions, avec "n" vecteurs différents dedans</content>
</new_definition>
<new_definition name="hilbert_space">
    <content>un espace de Hilbert est un espace vectoriel réel (ou complexe) muni d'un produit scalaire approprié, avec un nombre quelconque (fini ou infini) de dimensions</content>
</new_definition>
<new_definition name="prehilbert_space">
    <content>un espace préhilbertien est un espace de Hilbert, qui n'est pas forcément complet</content>
</new_definition>

<new_definition name="exhaustion_method">
    <content>la méthode d'exhaustion est une méthode (obsolète) de calcul d'une aire / d'un volume en se rapprochant successivement de cette valeur, par comblement du vide (ou retrait de matière) dans l'endroit calculé par des objets d'aire / volume connu</content>
</new_definition>
